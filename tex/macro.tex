\definecolor{darkgreen}{RGB}{0,50,0}
\newcommand\colornote[2]{\mbox{}\marginpar{\footnotesize\raggedright\hspace{0pt}\color{#1}{#2}}}
\newcounter{correctioncolor}[page]
\newcommand\correction[2]{%
	\underline{#1}%
	\mbox{}%
	\marginpar{%
		\footnotesize%
		\raggedright%
		\hspace{0pt}%
		\ifthenelse%
			{\isodd{\value{correctioncolor}}}%
			{\color{green}}%
			{\color{darkgreen}}%
		\addtocounter{correctioncolor}{1}%
		{\bf #2}%
}}

\newcounter{todocolor}[page]
\newcommand\todo[1]{%
	\mbox{}%
	\marginpar{%
		\footnotesize%
		\raggedright%
		\hspace{0pt}%
		\ifthenelse%
			{\isodd{\value{todocolor}}}%
			{\color{blue}}%
			{\color{red}}%
		\addtocounter{todocolor}{1}%
		{\bf #1}%
}%
}
